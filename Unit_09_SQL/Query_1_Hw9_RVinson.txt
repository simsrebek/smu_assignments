--employee information from employees table and using a left join from salaries to inlcude salary information


SELECT e.emp_no AS employee_number, 
       e.last_name, 
       e.first_name, 
       e.gender, 
       s.salary 
FROM   employees AS e 
       LEFT JOIN salaries AS s 
              ON e.emp_no = s. emp_no 
ORDER  BY e.last_name ASC, 
          e.first_name ASC;
